1    package com.example.budgetapp
2    
3    import android.content.Context
4    import android.content.Intent
5    import android.net.Uri
6    import android.os.Bundle
7    import android.util.Patterns
8    import android.widget.Button
9    import android.widget.EditText
10   import android.widget.ImageView
11   import android.widget.Toast
12   import androidx.appcompat.app.AppCompatActivity
13   
14   // Login screen activity for user authentication
15   class Login : AppCompatActivity() {
16   
17       override fun onCreate(savedInstanceState: Bundle?) {
18           super.onCreate(savedInstanceState)
19   
20           // Load login UI layout
21           setContentView(R.layout.activity_login)
22   
23           // UI references for input fields and buttons
24           val etEmail      = findViewById<EditText>(R.id.etEmail)
25           val etPassword   = findViewById<EditText>(R.id.etPassword)
26           val btnLogin     = findViewById<Button>(R.id.btnLogin)
27           val btnGoToRegister = findViewById<Button>(R.id.btnGoToRegister)
28   
29           // Social media icons
30           val imgWhatsapp  = findViewById<ImageView>(R.id.imgWhatsapp)
31           val imgInstagram = findViewById<ImageView>(R.id.imgInstagram)
32           val imgFacebook  = findViewById<ImageView>(R.id.imgFacebook)
33   
34           // Open WhatsApp link in browser/app
35           imgWhatsapp.setOnClickListener {
36               startActivity(Intent(Intent.ACTION_VIEW, Uri.parse("https://wa.me/")))
37           }
38   
39           // Open Instagram link
40           imgInstagram.setOnClickListener {
41               startActivity(Intent(Intent.ACTION_VIEW, Uri.parse("https://www.instagram.com/")))
42           }
43   
44           // Open Facebook link
45           imgFacebook.setOnClickListener {
46               startActivity(Intent(Intent.ACTION_VIEW, Uri.parse("https://www.facebook.com/")))
47           }
48   
49           // Navigate to Register screen
50           btnGoToRegister.setOnClickListener {
51               startActivity(Intent(this, Register::class.java))
52           }
53   
54           // Login button click handler
55           btnLogin.setOnClickListener {
56   
57               // Get user input and remove extra spaces
58               val email    = etEmail.text.toString().trim()
59               val password = etPassword.text.toString().trim()
60   
61               // -------- VALIDATION SECTION --------
62   
63               // Check if email is empty
64               if (email.isEmpty()) {
65                   etEmail.error = "Please enter an email"
66                   return@setOnClickListener
67               }
68   
69               // Check if email format is valid
70               if (!Patterns.EMAIL_ADDRESS.matcher(email).matches()) {
71                   etEmail.error = "Invalid email address"
72                   return@setOnClickListener
73               }
74   
75               // Check if password is empty
76               if (password.isEmpty()) {
77                   etPassword.error = "Please enter a password"
78                   return@setOnClickListener
79               }
80   
81               // Check password length requirement
82               if (password.length < 6) {
83                   etPassword.error = "you need to enter 6 characters"
84                   return@setOnClickListener
85               }
86   
87               // -------- AUTHENTICATION SECTION --------
88   
89               // Retrieve saved user data from SharedPreferences
90               val sharedPref = getSharedPreferences("UserPrefs", Context.MODE_PRIVATE)
91               val savedPassword = sharedPref.getString(email, null)
92   
93               // Check if user exists
94               if (savedPassword == null) {
95                   Toast.makeText(this, "User not found. Please register.", Toast.LENGTH_SHORT).show()
96   
97                   // Check if password matches saved password
98               } else if (savedPassword == password) {
99                   Toast.makeText(this, "Login Successful!", Toast.LENGTH_SHORT).show()
100  
101                  // Navigate to Dashboard after successful login
102                  startActivity(Intent(this, Dashboard::class.java))
103                  finish()
104  
105                  // Wrong password case
106              } else {
107                  Toast.makeText(this, "wrong password", Toast.LENGTH_SHORT).show()
108              }
109          }
110      }
111  }
