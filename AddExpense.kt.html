1    package com.example.budgetapp
2    
3    import android.os.Bundle
4    import android.widget.Button
5    import android.widget.EditText
6    import android.widget.Toast
7    import androidx.activity.enableEdgeToEdge
8    import androidx.appcompat.app.AppCompatActivity
9    import androidx.core.view.ViewCompat
10   import androidx.core.view.WindowInsetsCompat
11   import android.content.Intent
12   
13   class AddExpense : AppCompatActivity() {
14   
15       override fun onCreate(savedInstanceState: Bundle?) {
16           super.onCreate(savedInstanceState)
17           // Enable edge-to-edge display
18           enableEdgeToEdge()
19           setContentView(R.layout.activity_add_expense)
20   
21           // Set padding for system bars
22           ViewCompat.setOnApplyWindowInsetsListener(findViewById(R.id.main)) { v, insets ->
23               val systemBars = insets.getInsets(WindowInsetsCompat.Type.systemBars())
24               v.setPadding(systemBars.left, systemBars.top, systemBars.right, systemBars.bottom)
25               insets
26           }
27   
28           val etName = findViewById<EditText>(R.id.etExpenseName)
29           val etAmount = findViewById<EditText>(R.id.etExpenseAmount)
30           val btnSave = findViewById<Button>(R.id.btnSaveExpense)
31   
32           btnSave.setOnClickListener {
33               val name = etName.text.toString().trim()
34               val amountStr = etAmount.text.toString().trim()
35   
36               if (name.isEmpty()) {
37                   etName.error = "Please enter a name"
38                   return@setOnClickListener
39               }
40   
41               if (amountStr.isEmpty()) {
42                   etAmount.error = "Please enter an amount"
43                   return@setOnClickListener
44               }
45   
46               val amount = amountStr.toDoubleOrNull()
47               if (amount == null || amount <= 0) {
48                   etAmount.error = "Enter a valid amount"
49                   return@setOnClickListener
50               }
51   
52               // Prepare data to send back
53               val resultIntent = Intent()
54               resultIntent.putExtra("expenseName", name)
55               resultIntent.putExtra("expenseAmount", amount)
56   
57               setResult(RESULT_OK, resultIntent)
58               Toast.makeText(this, "Expense saved!", Toast.LENGTH_SHORT).show()
59   
60               // Finish activity and return data
61               finish()
62           }
63       }
64   }
